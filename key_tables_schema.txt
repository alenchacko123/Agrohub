Table: rental_requests
  id - int(11)
  equipment_id - int(11)
  equipment_name - varchar(255)
  farmer_id - int(11)
  farmer_name - varchar(255)
  farmer_email - varchar(255)
  owner_id - int(11)
  start_date - date
  end_date - date
  num_days - int(11)
  total_amount - decimal(10,2)
  delivery_address - text
  need_operator - tinyint(1)
  need_insurance - tinyint(1)
  special_requirements - text
  status - varchar(50)
  created_at - timestamp
  updated_at - timestamp
  agreement_status - varchar(50)
  is_notification_dismissed - tinyint(1)
  is_dismissed - tinyint(1)
